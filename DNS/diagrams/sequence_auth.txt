@startuml
actor Пользователь
participant "Окно авторизации" as UI
participant "Контроллер авторизации" as Auth
participant "База данных" as DB

Пользователь -> UI: Открытие окна авторизации
activate UI

Пользователь -> UI: Ввод логина и пароля
Пользователь -> UI: Нажатие кнопки "Войти"
UI -> Auth: Передача данных

activate Auth
Auth -> DB: Запрос проверки данных
activate DB

DB --> Auth: Результат проверки
deactivate DB

alt Успешная авторизация
    Auth --> UI: Подтверждение
    UI --> Пользователь: Переход в главное меню
else Ошибка авторизации
    Auth --> UI: Сообщение об ошибке
    UI --> Пользователь: Показ сообщения об ошибке
end

deactivate Auth
deactivate UI
@enduml
